The dataset:¶

The activation vectors of most neural networks can be represented as linear combinations of the neurons' output without loss of information. This means that information in the activations are generally not compositional. An example of information that is compositional would be the bits in a binary representation of an integer; the activation of one bit is informative in the context of another bit.

It would be interesting to see if neural nets can represent information compositionally, so let's try and train one to convert between integers and binary. If the network learns this representation, we should be able to obtain binary representations of examples that we withold from the training set.

Specifics. We consider the integers in range 0-32 (we need 5 bits to express them). Instead of feeding in raw integers, we express them as float values and divide by 32, meaning that they live in (0,1). Values that we hold out are {10, 25}. I chose these (semi-randomly) because they don't have about the same number of 0's and 1's in binary, and are not adjacent on the number line.

def onehot(values, n_values): # converts to one-hot (use np.argmax to convert back)
    oh = np.zeros((values.size, n_values))
    oh[range(values.size), values.ravel()] = 1
    return oh.reshape(list(values.shape) + [n_values])

class Dataloader():
    def __init__(self):
        self.vmax = 32.
        self.val_set = np.asarray([10.,25.])
        train_list = list(set(np.arange(self.vmax)) - set(self.val_set))
        self.train_set = np.asarray( train_list )
        
    def next(self, batch_size, train=True):
        dataset = self.train_set if train else self.val_set
        
        ixs = np.random.randint(len(dataset), size=batch_size)
        samples = dataset[ixs]
        
        X = np.asarray( samples / self.vmax ).reshape(batch_size, 1) #self.onehot(nums, self.vmax)
        y = np.asarray( [self.int2bin(s) for s in samples] )
        return X, y
    
    @staticmethod
    def int2bin(num):
        return [int(c) for c in list("{0:b}".format(int(num)).zfill(5))]
    
    @staticmethod
    def onehot(values, n_values): # converts to one-hot (use np.argmax to convert back)
        oh = np.zeros((values.size, n_values))
        oh[range(values.size), values.ravel()] = 1
        return oh.reshape(list(values.shape) + [n_values])
    
def accuracy(model, dataloader):
    X, y = dataloader.next(1000)
    model.eval() 
    y_hat = model(X)
    model.train()

    N = y_hat.shape[0] ; acc = 0
    for i in range(N):
        acc += np.allclose(y_hat[i], y[i], atol=0.2) / N # all values are within 0.2 from the correct bit
    return acc

class DenseClassifier(nn.Module):
    def __init__(self, xdim, ydim, hdim=20):
        super(DenseClassifier, self).__init__()
        self.linear1 = nn.Linear(xdim, hdim)
        self.linear2 = nn.Linear(hdim, hdim)
        self.linear3 = nn.Linear(hdim, ydim)

    def forward(self, x):
        x = Variable(torch.Tensor(x)) if type(x) == np.ndarray else x # maybe convert to an autograd variable
        h1 = F.relu(self.linear1(x))  # shape [batch x 28^2] -> [batch x hdim]
        h2 = F.relu(self.linear2(h1)) # shape [batch x hdim] -> [batch x hdim]
        h3 = F.sigmoid(self.linear3(h2)) # shape [batch x hdim] -> [batch x 10  ]
        return h3 if self.training else h3.data.detach().numpy() # maybe return as a numpy array, else as a Variable

for step in range(info.global_step+1, info.total_steps):
    
    X, y = dataloader.next(info.batch_size)  # get a minibatch
    y_hat = model(X)                        # forward pass
    
    loss = F.binary_cross_entropy(input=y_hat, target=torch.Tensor(y))    # this is the loss function
    loss.backward() ; optimizer.step() ; optimizer.zero_grad() # get grads, apply with optimizer
    
    info.loss += [loss.data.view(-1)[0]]
    run_loss = running_mean(run_loss, loss.data.view(-1)[0])
    info.global_step = step
    if time.time() - print_t > .2:
        print_t = time.time()
